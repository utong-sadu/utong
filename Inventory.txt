                const price = parseFloat(unitPrice) || 0;
                return qty * price;
            };

            const handleItemChange = (id, field, value) => {
                setItems(items.map(item => {
                    if (item.id === id) {
                        const updated = { ...item, [field]: value };
                        if (field === 'issuedQty' || field === 'unitPrice') {
                            updated.total = calculateTotal(updated.issuedQty, updated.unitPrice);
                        }
                        return updated;
                    }
                    return item;
                }));
            };

            const addItem = () => {
                const newId = Math.max(...items.map(i => i.id), 0) + 1;
                setItems([...items, {
                    id: newId,
                    referenceNumber: '',
                    name: '',
                    unit: '',
                    stockQty: '',
                    requestedQty: '',
                    issuedQty: '',
                    unitPrice: '',
                    total: 0
                }]);
            };

            const removeItem = (id) => {
                if (items.length > 1) {
                    setItems(items.filter(item => item.id !== id));
                }
            };

            const grandTotal = items.reduce((sum, item) => sum + item.total, 0);

            const handleSave = () => {
                const data = {
                    ...formData,
                    items: items,
                    grandTotal: grandTotal
                };
                console.log('Saving issue data:', data);
                alert('ទិន្នន័យត្រូវបានរក្សាទុក! (Data saved!)');
            };

            return (
                <div>
                    <div className="form-header">
                        <Truck />
                        <div className="form-title">កត់ត្រាប័ណ្ណបញ្ចេញសម្ភារៈ ទំនិញ</div>
                    </div>
                    
                    <div className="form-grid">
                        <div className="form-group">
                            <label>លេខប័ណ្ណបញ្ចេញសម្ភារៈ ទំនិញ *</label>
                            <select
                                value={formData.receiptNumber}
                                onChange={(e) => setFormData({...formData, receiptNumber: e.target.value})}
                                className="form-control"
                            >
                                {Array.from({ length: 20 }, (_, i) => {
                                    const num = String(i + 1).padStart(3, '0');
                                    return <option key={num} value={num}>{num}</option>;
                                })}
                            </select>
                        </div>

                        <div className="form-group">
                            <label>កាលបរិច្ឆេទបញ្ចេញសម្ភារៈ ទំនិញ *</label>
                            <input
                                type="date"
                                value={formData.receiptDate}
                                onChange={(e) => setFormData({...formData, receiptDate: e.target.value})}
                                className="form-control"
                            />
                        </div>

                        <div className="form-group">
                            <label>ឈ្មោះ និងផ្នែកប្រើប្រាស់  *</label>
                            <input
                                type="text"
                                value={formData.userName}
                                onChange={(e) => setFormData({...formData, userName: e.target.value})}
                                className="form-control"
                            />
                        </div>

                        <div className="form-group">
                            <label>មុខសញ្ញាប្រើប្រាស់ *</label>
                            <input
                                type="text"
                                value={formData.department}
                                onChange={(e) => setFormData({...formData, department: e.target.value})}
                                className="form-control"
                            />
                        </div>

                        <div className="form-group">
                            <label>បញ្ចេញនៅឃ្លាំង *</label>
                            <input
                                type="text"
                                value={formData.storageLocation}
                                onChange={(e) => setFormData({...formData, storageLocation: e.target.value})}
                                className="form-control"
                            />
                        </div>
                    </div>

                    <div className="table-container">
                        <table>
                            <thead>
                                <tr>
                                    <th>ល.រ</th>
                                    <th>យោង​បណ្ណបញ្ចូល *</th>
                                    <th>ឈ្មោះសញ្ញា ក្បួនខ្នាត បរិក្ខារ ទំនិញ *</th>
                                    <th>ឯកតាគិត</th>
                                    <th>ចំនួននៅសល់</th>
                                    <th>ស្នើសុំបញ្ចេញលើកនេះ*</th>
                                    <th>បញ្ចេញពិតប្រាកដ *</th>
                                    <th>ថ្លៃឯកតា</th>
                                    <th>ទឹកប្រាក់សរុប</th>
                                    <th></th>
                                </tr>
                            </thead>
                            <tbody>
                                {items.map((item, index) => (
                                    <tr key={item.id}>
                                        <td>{index + 1}</td>
                                        <td>
                                            <input
                                                type="text"
                                                value={item.referenceNumber}
                                                onChange={(e) => handleItemChange(item.id, 'referenceNumber', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td>
                                            <input
                                                type="text"
                                                value={item.name}
                                                onChange={(e) => handleItemChange(item.id, 'name', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td>
                                            <input
                                                type="text"
                                                value={item.unit}
                                                onChange={(e) => handleItemChange(item.id, 'unit', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td>
                                            <input
                                                type="number"
                                                value={item.stockQty}
                                                onChange={(e) => handleItemChange(item.id, 'stockQty', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td>
                                            <input
                                                type="number"
                                                value={item.requestedQty}
                                                onChange={(e) => handleItemChange(item.id, 'requestedQty', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td>
                                            <input
                                                type="number"
                                                value={item.issuedQty}
                                                onChange={(e) => handleItemChange(item.id, 'issuedQty', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td>
                                            <input
                                                type="number"
                                                value={item.unitPrice}
                                                onChange={(e) => handleItemChange(item.id, 'unitPrice', e.target.value)}
                                                className="form-control"
                                            />
                                        </td>
                                        <td className="text-right">
                                            {item.total.toLocaleString()}
                                        </td>
                                        <td className="action-cell">
                                            <button
                                                onClick={() => removeItem(item.id)}
                                                className="btn btn-danger"
                                                disabled={items.length === 1}
                                            >
                                                <Trash2 />
                                            </button>
                                        </td>
                                    </tr>
                                ))}
                            </tbody>
                            <tfoot>
                                <tr className="total-row">
                                    <td colSpan="8" className="text-right">សរុប</td>
                                    <td className="text-right">{grandTotal.toLocaleString()}</td>
                                    <td></td>
                                </tr>
                            </tfoot>
                        </table>
                    </div>

                    <div className="btn-group">
                        <button
                            onClick={addItem}
                            className="btn btn-success"
                        >
                            <Plus />
                            បន្ថែមជួរ
                        </button>
                        <button
                            onClick={handleSave}
                            className="btn btn-primary"
                        >
                            <Save />
                            រក្សាទុក
                        </button>
                    </div>
                </div>
            );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
</body>
